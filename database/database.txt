CREATE TABLE IF NOT EXISTS business(businessid int NOT NULL AUTO_INCREMENT,businessname varchar(50),emailid varchar(20) NOT NULL,password varchar(20),website varchar(100),logo varchar(20),phone int(10),address varchar(200),location varchar(100),latlong varchar(50),idproof varchar(20),addressproof varchar(20),isapproved boolean DEFAULT false,flag boolean DEFAULT true,datecreated TIMESTAMP DEFAULT CURRENT_TIMESTAMP,PRIMARY KEY (businessid))
